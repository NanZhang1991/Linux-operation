pip config list # 查看当前镜像源
pip config set global.index-url https://pypi.douban.com/simple #更换镜像源
pip config set global.index-url https://pypi.tuna.tsinghua.edu.cn/simple
pip install package_name #安装包
pip install package_name ==1.0.4 # 指定版本
pip uninstall package_name #卸载包
pip show package_name #查看安装包信息
pip list  #列出所有安装包
pip list -outdated 检测更新 
pip install --upgrade package_name #更新包
pip freeze >environment.txt #对于不使用conda的用户，可以使用这段命令 将一个txt文件导出并包括在其中
pip install -r /path/environment.txt #以管理员身份运行cmd下的python命令，安装你刚导出来的environment.txt  /path/environment.txt导出的文件在本地的实际路径
